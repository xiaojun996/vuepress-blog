# 希尔排序

JavaScript 实现的希尔排序算法

<!-- more -->

### 代码

```JavaScript
/**
 * @param {Array} arr
 * @return {Array}
 */
const shellSort = arr => {
  const len = arr.length
  let gap = 1
  while (gap < len / 3) {
    gap = gap * 3 + 1
  }

  while (gap > 0) {
    // 注意下面这段 for 循环和插入排序极为相似
    for (let i = gap; i < len; i++) {
      const temp = arr[i]
      let preIndex = i - gap

      while (preIndex >= 0 && arr[preIndex] > temp) {
        arr[preIndex + gap] = arr[preIndex]
        preIndex -= gap
      }
      arr[preIndex + gap] = temp
    }
    gap = gap >> 1
  }

  return arr
}
```
