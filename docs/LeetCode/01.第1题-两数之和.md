/**
 * @param {Array} nums
 * @param {Array} target
 * @return {Array}
 */
let twoSum = function (nums, target) {
  let array = nums
  let count = target
  const length = array.length
  let a = []

  for (let i = 0; i < length - 1; i++) {
    for (let j = 1; j < length; j++) {
      if (array[i] + array[j] === count) {
        if (i !== j) {
          a[0] = i
          a[1] = j
          return a
        }
      }
    }
  }
  return 0
}
