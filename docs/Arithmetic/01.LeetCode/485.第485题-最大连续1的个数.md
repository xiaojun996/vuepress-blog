function findMaxConsecutiveOnes(nums: number[]): number {
  let count: number = 0
  let max: number = 0
  nums.forEach(v => {
    v === 0 ? (count = 0) : (count++)
    max = Math.max(max, count)
  })
  return max
}

impl Solution {
  pub fn find_max_consecutive_ones(nums: Vec<i32>) -> i32 {
    let mut count = 0;
    let mut max_size = 0;
    for i in 0..nums.len() {
      if nums[i] == 0 {
        count = 0;
      } else {
        count += 1;
      }
      max_size = max_size.max(count)
    }
    return max_size;
  }
}
